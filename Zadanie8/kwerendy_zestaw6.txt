1.
SELECT CLIENT.FULL_NAME, CLIENT_ADDRESS.STREET, LENGTH(CLIENT_ADDRESS.STREET) AS "Dlugosc ulicy", LEFT(CLIENT_ADDRESS.STREET, 5) AS "Pierwsze liter" FROM CLIENT LEFT JOIN CLIENT_ADDRESS ON CLIENT.CLIENT_ID = CLIENT_ADDRESS.CLIENT_ID;

2.
SELECT CLIENT.SHORT_NAME, IF(CLIENT_ADDRESS.STREET IS NULL, 'BRAK ADRESU', 'ADRES PRZYPISANY') AS "Typ adresu" FROM CLIENT LEFT JOIN CLIENT_ADDRESS ON CLIENT.CLIENT_ID = CLIENT_ADDRESS.CLIENT_ID;

3.
SELECT PRODUCT.PRODUCT_NAME AS Nazwa,CATEGORY.CATEGORY_NAME AS "Nazwa Kategorii", LEFT(PRODUCT.PRODUCT_NAME, 3) FROM PRODUCT JOIN CATEGORY ON PRODUCT.CATEGORY_ID = CATEGORY.CATEGORY_ID;

4.
SELECT PRODUCT.PRODUCT_NAME, PRODUCT.DESCRIPTION FROM PRODUCT JOIN ORDER_POSITION ON PRODUCT.PRODUCT_ID = ORDER_POSITION.PRODUCT_ID;

Zad 2.

1.
SELECT CLIENT.FULL_NAME, `ORDER`.ORDER_ID, `ORDER`.ORDER_DATE, DATEDIFF(CURRENT_DATE, `ORDER`.ORDER_DATE) AS "Dni ktore uplynely" FROM `ORDER` JOIN CLIENT ON `ORDER`.CLIENT_ID = CLIENT.CLIENT_ID ORDER BY "Dni ktore uplynely" DESC;

2.
SELECT DATE_ADD(`ORDER`.ORDER_DATE, INTERVAL 14 DAY) AS TERMIN FROM `ORDER` JOIN ORDER_STATUS ON `ORDER`.ORDER_STATUS = ORDER_STATUS.STATUS_ID WHERE STATUS_NAME NOT IN ('Zwrocone', 'Zamkniete') ;

3.
SELECT EMPLOYEE.FIRST_NAME,EMPLOYEE.LAST_NAME, EMPLOYEE.DATE_OF_EMPLOYMENT, DAYNAME(EMPLOYEE.DATE_OF_EMPLOYMENT) AS "Dzien tygodnia", IFNULL(DEPARTMENT.DEPARTMENT_NAME, 'Brak dzialu') AS "Nazwa dzialu" FROM EMPLOYEE LEFT JOIN DEPARTMENT ON EMPLOYEE.DEPARTMENT_ID = DEPARTMENT.DEPARTMENT_ID;

4.
SELECT CLIENT.FULL_NAME, DATEDIFF(MAX(`ORDER`.ORDER_DATE),MIN(`ORDER`.ORDER_DATE)) AS Roznica  FROM `ORDER` JOIN CLIENT ON `ORDER`.CLIENT_ID = CLIENT.CLIENT_ID GROUP BY CLIENT.CLIENT_ID, CLIENT.FULL_NAME ORDER BY Roznica DESC;

Zad 3.

1.
SELECT CLIENT.SHORT_NAME, GROUP_CONCAT(CLIENT_ADDRESS.STREET SEPARATOR ', ') AS "Lista adresow" FROM CLIENT JOIN CLIENT_ADDRESS ON CLIENT.CLIENT_ID = CLIENT_ADDRESS.CLIENT_ID GROUP BY CLIENT.CLIENT_ID, CLIENT.SHORT_NAME ORDER BY CLIENT.SHORT_NAME;

2.
SELECT CITY, GROUP_CONCAT(DISTINCT SUBSTR(POSTAL_CODE, 1,2) ORDER BY SUBSTR(POSTAL_CODE, 1, 2) SEPARATOR ', ') AS Unikalne FROM CLIENT_ADDRESS GROUP BY CITY ORDER BY CITY;

3.
SELECT CONCAT(EMPLOYEE.LAST_NAME, ' ', EMPLOYEE.FIRST_NAME) AS "Nazwisko Imie", IFNULL(DEPARTMENT.DEPARTMENT_NAME, 'Brak dzialu') AS Dzial, DATEDIFF(CURDATE(), EMPLOYEE.DATE_OF_EMPLOYMENT) AS "Dni od zatrudnienia" FROM EMPLOYEE JOIN DEPARTMENT ON EMPLOYEE.DEPARTMENT_ID = DEPARTMENT.DEPARTMENT_ID ORDER BY EMPLOYEE.LAST_NAME;

4.
SELECT CONCAT(CLIENT.FULL_NAME, ' - ', CLIENT.SHORT_NAME) AS Nazwa,GROUP_CONCAT(LPAD(MOD(`ORDER`.ORDER_ID, 1000),3,'0') ORDER BY `ORDER`.ORDER_ID SEPARATOR ', ' ) AS "Ostatnie 3 cyfry" FROM CLIENT JOIN `ORDER` ON CLIENT.CLIENT_ID = `ORDER`.CLIENT_ID GROUP BY CLIENT.CLIENT_ID, CLIENT.FULL_NAME, CLIENT.SHORT_NAME ORDER BY CLIENT.FULL_NAME;

5.
SELECT FIRST_NAME, LAST_NAME, BIRTH_DATE, CONCAT(LOWER(LEFT(FIRST_NAME, 3)), UPPER(RIGHT(LAST_NAME, 4)),(LENGTH(FIRST_NAME) + YEAR(BIRTH_DATE) + MONTH(BIRTH_DATE) + DAY(BIRTH_DATE))) AS Haslo FROM EMPLOYEE;

Zad 4.

1.
SELECT ORDER_DATE AS "Oryginalna data", DATE_FORMAT(ORDER_DATE, '%d-%m-%Y') AS "Format 1", DATE_FORMAT(ORDER_DATE, '%Y - %u') AS 'Format 2', DATE_FORMAT(ORDER_DATE, '%Y - %j') AS 'Format 3', DATE_FORMAT(ORDER_DATE, '%M/%d/%Y') AS 'Format 4' FROM `ORDER`;

2.
SELECT DATE_FORMAT(o.ORDER_DATE, '%u') AS numer_tygodnia, SUM(op.AMOUNT * op.SELL_PRICE) AS laczna_wartosc FROM `ORDER` o JOIN ORDER_POSITION op ON o.ORDER_ID = op.ORDER_ID WHERE YEAR(o.ORDER_DATE) = 2025 GROUP BY DATE_FORMAT(o.ORDER_DATE, '%u') ORDER BY laczna_wartosc DESC LIMIT 1;

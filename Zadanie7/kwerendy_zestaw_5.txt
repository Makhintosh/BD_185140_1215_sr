SELECT PRODUCT.PRODUCT_NAME AS nazwa_produktu, AVG(ORDER_POSITION.SELL_PRICE - PRODUCT.PURCHASE_PRICE) AS srednia_roznica FROM PRODUCT JOIN ORDER_POSITION ON PRODUCT.PRODUCT_ID = ORDER_POSITION.PRODUCT_ID GROUP BY PRODUCT.PRODUCT_
ID, PRODUCT.PRODUCT_NAME ORDER BY srednia_roznica DESC;

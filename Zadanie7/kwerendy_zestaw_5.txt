SELECT CLIENT.* FROM CLIENT LEFT JOIN `ORDER` ON CLIENT.CLIENT_ID = `ORDER`.CLIENT_ID WHERE `ORDER`.ORDER_ID IS NULL;
SELECT * FROM CLIENT WHERE CLIENT_ID NOT IN (SELECT CLIENT_ID FROM `ORDER`);

2.
SELECT UNIT_OF_MEASUREMENT.UNIT_NAME FROM INVENTORY RIGHT JOIN UNIT_OF_MEASUREMENT ON INVENTORY.UNIT_ID = UNIT_OF_MEASUREMENT.UNIT_ID IS NULL;

3.
SELECT PRODUCT.PRODUCT_NAME AS nazwa_produktu, AVG(ORDER_POSITION.SELL_PRICE - PRODUCT.PURCHASE_PRICE) AS srednia_roznica FROM PRODUCT JOIN ORDER_POSITION ON PRODUCT.PRODUCT_ID = ORDER_POSITION.PRODUCT_ID GROUP BY PRODUCT.PRODUCT_
ID, PRODUCT.PRODUCT_NAME ORDER BY srednia_roznica DESC;

Zad 2.

1.
SELECT DISTINCT D.DEPARTMENT_NAME FROM DEPARTMENT D JOIN EMPLOYEE E ON D.DEPARTMENT_ID = E.DEPARTMENT_ID JOIN
`ORDER` O ON E.EMPLOYEE_NAME = O.EMPLOYEE_ID;

2.
SELECT D.DEPARTMENT_NAME AS nazwa_departamentu, COUNT(O.ORDER_NUMBER) AS loczna_wartosc FROM DEPARTMENT D JOIN
EMPLOYEE E ON D.DEPARTMENT_ID = E.DEPARTMENT_ID JOIN `ORDER` O ON E.EMPLOYEE_NAME = O.EMPLOYEE_ID GROUP BY D.DEPARTMENT_NAME;

Zad 3.

1.
SELECT CONCAT(PRODUCT_NAME, ' (', DESCRIPTION, ')') AS Pelna_nazwa FROM PRODUCT;

2.
SELECT CONCAT_WS(',', PRODUCT_ID, PRODUCT_NAME, DESCRIPTION, CATEGORY_ID) AS Produkt FROM PRODUCT;

3.
SELECT DEPARTMENT.DEPARTMENT_NAME, GROUP_CONCAT(CONCAT(EMPLOYEE.FIRST_NAME, ' ', EMPLOYEE.LAST_NAME) SEPARATOR ', ') AS "Lista pracownikow" FROM DEPARTMENT JOIN EMPLOYEE ON DEPARTMENT.DEPARTMENT_ID = EMPLOYEE.DEPARTMENT_ID GROUP BY DEPARTMENT.DEPARTMENT_NAME ORDER BY DEPARTMENT.DEPARTMENT_NAME;

4.
SELECT CATEGORY.CATEGORY_NAME, GROUP_CONCAT(PRODUCT.PRODUCT_NAME SEPARATOR ' | ') AS "Produkty niesprzedane" FROM CATEGORY JOIN PRODUCT ON CATEGORY.CATEGORY_ID = PRODUCT.CATEGORY_ID WHERE PRODUCT.PRODUCT_ID NOT IN (SELECT DISTINCT PRODUCT_ID FROM ORDER_POSITION) GROUP BY CATEGORY.CATEGORY_NAME ORDER BY CATEGORY.CATEGORY_NAME;

Zad 4.

1.
SELECT ORDER_ID, IF(ORDER_STATUS IN (3, 4, 7, 10), 'Zakonczone', 'Inny') AS STATUSopis FROM `ORDER`;

2.
SELECT PRODUCT.PRODUCT_NAME, CASE WHEN INVENTORY.AMOUNT = 0 THEN 'BRAK' WHEN INVENTORY.AMOUNT > 0 AND INVENTORY.AMOUNT < 10 THEN 'Mala ilosc' WHEN INVENTORY.AMOUNT >= 10 AND INVENTORY.AMOUNT <= 30 THEN 'Srednia ilosc' WHEN INVENTORY.AMOUNT > 30 THEN 'Duza ilosc' ELSE 'Brak danych' END AS Dostepnosc FROM PRODUCT JOIN INVENTORY ON PRODUCT.PRODUCT_ID = INVENTORY.PRODUCT_ID ORDER BY PRODUCT.PRODUCT_NAME;

3.
SELECT CLIENT_ID, IFNULL(TAX_IDENTIFIER, 'KLIENT indywidualny') AS "Identyfikator klienta" FROM CLIENT;

Zad 5.

1.
SELECT CITY, SUBSTR(POSTAL_CODE, 1, 2) AS "Kod pocztowy" FROM CLIENT_ADDRESS;

2.
SELECT CITY, GROUP_CONCAT(DISTINCT SUBSTR(POSTAL_CODE, 1, 2) ORDER BY SUBSTR(POSTAL_CODE, 1, 2) SEPARATOR ', ')AS Unikalne FROM CLIENT_ADDRESS GROUP BY CITY ORDER BY CITY;

SELECT TRIM(IF(LOWER(STREET) LIKE 'ul. %', SUBSTR(STREET, 5), IF(LOWER(STREET) LIKE 'al. %', SUBSTR(STREET, 5),STREET)))AS "Ulica bez numeru", STREET AS "Pelny adres" FROM CLIENT_ADDRESS;

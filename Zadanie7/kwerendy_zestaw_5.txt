SELECT JOB_POSITION.POSITION_NAME, EMPLOYEE.FIRST_NAME, EMPLOYEE.LAST_NAME, EMPLOYEE.BIRTH_DATE, DEPARTMENT.DEPARTMENT_NAME
FROM EMPLOYEE
JOIN DEPARTMENT ON EMPLOYEE.DEPARTMENT_ID = DEPARTMENT.DEPARTMENT_ID
JOIN JOB_POSITION ON JOB_POSITION.JOB_POSITION_ID = EMPLOYEE.JOB_POSITION_ID
WHERE EMPLOYEE.BIRTH_DATE = (SELECT MAX(BIRTH_DATE) FROM EMPLOYEE WHERE JOB_POSITION_ID = JOB_POSITION.JOB_POSITION_ID);
